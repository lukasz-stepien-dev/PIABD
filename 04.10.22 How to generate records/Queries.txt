1. Wypisz wszystkich agentów którzy w ostatnim roku sprzedali polisę

SELECT agenci.imie, agenci.nazwisko, agenci.id 
FROM agenci, ubezpieczenia, skladki WHERE
agenci.id = ubezpieczenia.ag_id AND
ubezpieczenia.skladka = skladki.ub_polisa AND
skladki.data_oplacenia < '2022-10-04' AND
skladki.data_oplacenia > '2021-10-04';
	
2. Wypisz wszystkie osoby które kupiły polisę w miesiącach parzystych

SELECT osoby.nazwisko, osoby.imie, osoby.miasto, osoby.id
FROM osoby, ubezpieczenia, skladki WHERE
osoby.id = ubezpieczenia.os_id AND
ubezpieczenia.skladka = skladki.ub_polisa AND
MONTH(skladki.miesiac) % 2 = 0;
	
3. Wypisz wszystkie składki których termin płatności jest pomiedzy 10.04.2023 i 10.04.2024 wraz z informacją u którego agenta  były wykupione

SELECT skladki.ub_polisa, skladki.termin_platnosci, agenci.id, agenci.imie, agenci.nazwisko
FROM skladki, ubezpieczenia, agenci WHERE
skladki.ub_polisa = ubezpieczenia.skladka AND
ubezpieczenia.ag_id = agenci.id AND
skladki.termin_platnosci BETWEEN '2023-10-04' AND '2024-10-04';
	
4. Wypisz wszystkie osoby które są z miasta Owando i mają nazwisko na K, B lub M

SELECT *
FROM osoby WHERE
miasto = 'Owando' AND
(nazwisko LIKE 'K%' OR
nazwisko LIKE 'B%' OR
nazwisko LIKE 'M%');
	
5. Wypisz wszystkie osoby które mają inicjały KA lub KM

SELECT * FROM osoby
WHERE imie LIKE 'K%' AND 
(nazwisko LIKE 'A%' OR nazwisko LIKE 'M%');
	
6. Wypisz wszystkich agentów wraz z informacją jaki wariant ubezpieczenia sprzedają

SELECT agenci.id, agenci.imie, agenci.nazwisko, ubezpieczenia.wariant 
FROM agenci, ubezpieczenia WHERE
agenci.id = ubezpieczenia.ag_id;
	
7. Stwórz 5 własnych zapytań łączących minimum 3 tabele

	- Wyswietl agentow, wariant ubezpieczenia i osoby, które pochodzą z miast na litera A

	SELECT agenci.imie, agenci.nazwisko, ubezpieczenia.wariant, osoby.imie, osoby.nazwisko, osoby.miasto
	FROM agenci, ubezpieczenia, osoby WHERE
	agenci.id = ubezpieczenia.ag_id AND
	ubezpieczenia.os_id = osoby.id AND
	osoby.miasto LIKE 'A%';

	- Wyświetl agentów, data_od i data_oplacenia od 2018.

	 SELECT agenci.imie, agenci.nazwisko, ubezpieczenia.data_od, skladki.data_oplacenia
	FROM agenci, ubezpieczenia, skladki WHERE
	agenci.id = ubezpieczenia.ag_id AND
	ubezpieczenia.skladka = skladki.ub_polisa AND
	skladki.data_oplacenia >= '2018-01-01';

	- Wyświetl Miasto, imie i nazwisko agenta jak i data_do, gdzie wariant to 7.

	SELECT agenci.imie, agenci.nazwisko, osoby.miasto, ubezpieczenia.data_do 
	FROM agenci, ubezpieczenia, osoby WHERE
	agenci.id = ubezpieczenia.ag_id AND
	ubezpieczenia.os_id = osoby.id AND
	ubezpieczenia.wariant = 7;

	- Wyświetl agentów i miasta zaczynające się na F lub kończące się na  Z.

	SELECT agenci.imie, agenci.nazwisko, osoby.miasto 
	FROM agenci, ubezpieczenia, osoby WHERE
	agenci.id = ubezpieczenia.ag_id AND
	ubezpieczenia.os_id = osoby.id AND
	(osoby.miasto LIKE 'F%' OR
	osoby.miasto LIKE 'Z%');

	- Wyświetl osoby i skladki, które mają termin płatności do 2025

	SELECT osoby.id, osoby.nazwisko, osoby.imie, skladki.termin_platnosci 
	FROM skladki, ubezpieczenia, osoby WHERE
	skladki.ub_polisa = ubezpieczenia.skladka AND
	ubezpieczenia.os_id = osoby.id AND
	skladki.termin_platnosci BETWEEN '2022-10-04' AND '2025-01-01';
